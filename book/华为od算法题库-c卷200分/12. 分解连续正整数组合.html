# 12. 分解连续正整数组合

const readline = require('readline');
const rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout,
});

rl.on('line', (line) => {
    const n = parseInt(line);
    for(let m=2; m*(1+m)/2<=n; m++) {
        if((n - m*(m-1)/2)%m === 0) {
            let x = (n - m*(m-1)/2)/m;
            let res = `${n}=`;
            for(let i=0; i<m; i++) {
                res += (x + i).toString();
                if (i < m-1) {
                    res += '+';
                }
            }
            console.log(res);
            return;
        }
    }
    console.log('N');
});
