# CommonsChunkPlugin

CommonsChunkPluginжҳҜwebpack4д№ӢеүҚеҶ…йғЁиҮӘеёҰзҡ„жҸ’д»¶пјҲ4д№ӢеҗҺжӣҝжҚўдёәдәҶSplitChunksпјүгҖӮе®ғеҸҜд»Ҙе°ҶеӨҡдёӘChunkдёӯе…¬е…ұзҡ„йғЁеҲҶжҸҗеҸ–еҮәжқҘгҖӮ

  1. ејҖеҸ‘иҝҮзЁӢдёӯеҮҸе°‘дәҶйҮҚеӨҚжЁЎеқ—жү“еҢ…пјҢеҸҜд»ҘжҸҗеҚҮејҖеҸ‘йҖҹеәҰгҖӮ
  2. еҮҸе°Ҹж•ҙдҪ“иө„жәҗдҪ“з§ҜгҖӮ
  3. еҗҲзҗҶеҲҶзүҮеҗҺзҡ„д»Јз ҒеҸҜд»Ҙжӣҙжңүж•Ҳзҡ„еҲ©з”Ёе®ўжҲ·з«Ҝзј“еӯҳгҖӮ

    entry: {
        foo: './foo.js',
        bar: './bar.js',

            filename: 'commons.js' // иө„жәҗзҡ„ж–Үд»¶еҗҚ
        })
    ]
}

# Infinity

и®ҫзҪ®дёәж— з©·д»ЈиЎЁжҸҗеҸ–зҡ„еҖјж— зәҝй«ҳпјҢд№ҹе°ұжҳҜиҜҙжүҖжңүжЁЎеқ—йғҪдёҚдјҡиў«жҸҗеҸ–гҖӮ

  1. жҲ‘д»¬еҸӘжғіи®©webpackжҸҗеҸ–зү№е®ҡзҡ„еҮ дёӘжЁЎеқ—пјҢе№¶е°ҶиҝҷдәӣжЁЎеқ—йҖҡиҝҮж•°з»„еһӢе…ҘеҸЈдј е…ҘгҖӮ
  2. дёәдәҶз”ҹжҲҗдёҖдёӘжІЎжңүд»»дҪ•жЁЎеқ—иҖҢд»…д»…еҢ…еҗ«webpackеҲқе§ӢеҢ–зҺҜеўғзҡ„ж–Үд»¶пјҢиҝҷдёӘж–Үд»¶жҲ‘д»¬з§°дёәmanifestгҖӮ

# еҮҪж•°

webpackжү“еҢ…иҝҮзЁӢдёӯзҡ„жҜҸдёӘжЁЎеқ—йғҪдјҡз»ҸиҝҮиҝҷдёӘеҮҪж•°зҡ„еӨ„зҗҶпјҢеҪ“еҮҪж•°зҡ„иҝ”еӣһеҖјдёәtrueж—¶иҝӣиЎҢжҸҗеҸ–гҖӮ

new webpack.optimize.CommonsChunkPlugin({
    name: 'vendor',
    filename: 'vendor.js',
    minChunks: function(module, count) {
        // module.context жЁЎеқ—зӣ®еҪ•и·Ҝеҫ„
        if(module.context && module.context.includes('mode_modules')) {
            return true;
        }
        
        // module.resource еҢ…еҗ«жЁЎеқ—еҗҚзҡ„е®Ңж•ҙи·Ҝеҫ„
        if(module.resource && module.resource.endsWith('util.js')) {
            return true;
        }
        
        // count дёәжЁЎеқ—иў«еј•з”Ёзҡ„ж¬Ўж•°
        if(count > 5) {
            return true;
        }
    }
})

# hashдёҺй•ҝж•Ҳзј“еӯҳ

CommonsChunkPluginжҸҗеҸ–е…¬е…ұжЁЎеқ—ж—¶пјҢжҸҗеҸ–еҗҺзҡ„иө„жәҗеҶ…йғЁдёҚд»…д»…жҳҜжЁЎеқ—зҡ„д»Јз ҒпјҢеҫҖеҫҖиҝҳеҢ…еҗ«webpackзҡ„иҝҗиЎҢж—¶гҖӮжҢҮзҡ„жҳҜеҲқе§ӢеҢ–зҺҜеўғзҡ„д»Јз ҒпјҢеҰӮеҲӣе»әжЁЎеқ—зј“еӯҳеҜ№иұЎгҖҒеЈ°жҳҺжЁЎеқ—еҠ иҪҪеҮҪж•°зӯүгҖӮ

ж—©жңҹwebpackдёӯпјҢиҝҗиЎҢж—¶еҶ…йғЁд№ҹеҢ…еҗ«жЁЎеқ—idпјҢе№¶дё”иҝҷдёӘidжҳҜд»Ҙж•°еӯ—дёҚж–ӯзҙҜеҠ зҡ„гҖӮиҝҷдјҡеҜјиҮҙдёҖдёӘй—®йўҳпјҢжЁЎеқ—idзҡ„ж”№еҸҳдјҡеҜјиҮҙиҝҗиЎҢж—¶еҶ…йғЁзҡ„д»Јз ҒеҸ‘з”ҹеҸҳеҢ–пјҢиҝӣдёҖжӯҘеҪұе“Қchunk hashзҡ„з”ҹжҲҗгҖӮдёҖиҲ¬дјҡдҪҝз”Ёchunk hashдҪңдёәиө„жәҗзҡ„зүҲжң¬еҸ·дјҳеҢ–е®ўжҲ·з«Ҝзј“еӯҳпјҢзүҲжң¬еҸ·ж”№еҸҳдјҡеҜјиҮҙз”ЁжҲ·йў‘з№Ғзҡ„жӣҙж–°иө„жәҗпјҢеҚідҪҝе®ғд»¬зҡ„еҶ…е®№е№¶жІЎжңүеҸ‘з”ҹеҸҳеҢ–д№ҹдјҡжӣҙж–°гҖӮгҖӮ

  3. з”ұдәҺеҶ…йғЁи®ҫи®Ўзјәйҷ·пјҢеңЁжҸҗеҸ–е…¬е…ұжЁЎеқ—зҡ„ж—¶еҖҷдјҡз ҙеқҸжҺүеҺҹжңүChunkдёӯжЁЎеқ—зҡ„дҫқиө–е…ізі»пјҢеҜјиҮҙйҡҫд»ҘиҝӣиЎҢжӣҙеӨҡзҡ„дјҳеҢ–гҖӮжҜ”еҰӮеңЁејӮжӯҘChunkзҡ„еңәжҷҜдёӢCommonsChunkPluginе№¶дёҚдјҡжҢүз…§йў„жңҹжӯЈеёёе·ҘдҪңгҖӮ
