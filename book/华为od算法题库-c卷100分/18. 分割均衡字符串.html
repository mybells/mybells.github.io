# 18. 分割均衡字符串

const readline = require('readline');
const rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout,
});
rl.on('line', function(str)) {
    let ans = 0;
    let count = 0;
    for(let i=0; i<str.length; i++) {
        if (str[i] === 'X') {
            count++;
        } else {
            count--;
        }
        if (count === 0) {
            ans++;
        }
    }
    console.log(ans);
}
