# 66. 密码解密

const readline = require('readline');
const rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout,
});

rl.on('line', function(s) {
    for(let i=26; i>=1; i--) {
        const key = i + (i >= 10 ? '*' : '');
        const value = String.fromCharCode(96 + i);
        s = s.replaceAll(key, value);
    }
    console.log(s);
});
